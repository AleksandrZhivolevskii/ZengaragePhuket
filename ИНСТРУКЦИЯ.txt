ZEN GARAGE PHUKET — Финальная версия для Netlify
=====================================================

ЭТА ВЕРСИЯ ИСПРАВЛЯЕТ ОШИБКУ "Page not found"
Используется файл _redirects (надёжнее чем netlify.toml redirects
для Drag&Drop деплоя).

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
ШАГ 1 — Загрузить на Netlify
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Распакуйте этот ZIP — получится папка ZenGarageReady
2. Зайдите на ваш существующий сайт в Netlify → вкладка "Deploys"
3. Перетащите ВСЮ папку ZenGarageReady целиком
   (внутри должны быть видны: dist/, netlify/, netlify.toml,
   package.json — все вместе)
4. Дождитесь окончания деплоя (~30-60 секунд)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
ШАГ 2 — Проверить DATABASE_URL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Site configuration → Environment variables
2. Убедитесь что есть переменная:
   Key:   DATABASE_URL
   Value: postgresql://neondb_owner:...@...neon.tech/neondb?sslmode=require
   
3. Если переменной нет — добавьте её, затем:
   Deploys → Trigger deploy → Deploy site

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
ШАГ 3 — Проверка
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Откройте ваш сайт (zen-garage-phuket.netlify.app)
2. Должен появиться загрузочный экран → потом календарь
3. Запишите тестового клиента в любой слот
4. Обновите страницу (F5 или Cmd+R)
5. Если запись осталась на месте — всё работает правильно ✅

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
ЕСЛИ ВСЁ ЕЩЁ "Page not found"
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Проверьте Site configuration → Build & deploy → 
"Publish directory" — должно быть указано "dist"
(не "ZenGarageReady/dist", а именно "dist")

Если там написано что-то другое — нажмите Edit settings,
впишите "dist", сохраните, затем Deploys → Trigger deploy.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
ЕСЛИ ЗАПИСИ НЕ СОХРАНЯЮТСЯ (но 404 нет)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Откройте сайт → нажмите F12 → вкладка "Console"
2. Попробуйте записать клиента
3. Если видите красную ошибку — сделайте скриншот и пришлите

Частая причина: DATABASE_URL не добавлен или добавлен с опечаткой.
Сверьте строку с той что в Neon Dashboard → Connection string.
